🔥 Metric Registry (Canonical Design)

The registry is a typed, versioned, hierarchical store with four layers:

1. Metric Node Registry — atomic metric definitions  
2. Metric Pack Registry — grouped metrics (identity, semantic, variant, etc.)  
3. Metric DAG Registry — executable graphs built from nodes  
4. Metric Profile Registry — user‑facing presets (e.g., “deep semantic”, “fast identity”)  

Below is the full JSON schema for the registry, followed by a combined registry artifact containing all previously defined metrics.

📘 Metric Registry JSON Schema

`json
{
  "$id": "https://example.com/schema/metric-registry.json",
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "title": "Metric Registry",
  "type": "object",
  "required": ["version", "nodes", "packs", "dags", "profiles"],
  "properties": {
    "version": {
      "type": "string",
      "description": "Semantic version of the registry."
    },

    "nodes": {
      "type": "array",
      "description": "Atomic metric node definitions.",
      "items": { "$ref": "#/$defs/MetricNode" }
    },

    "packs": {
      "type": "array",
      "description": "Logical groupings of metric nodes.",
      "items": { "$ref": "#/$defs/MetricPack" }
    },

    "dags": {
      "type": "array",
      "description": "Executable metric DAGs.",
      "items": { "$ref": "#/$defs/MetricDag" }
    },

    "profiles": {
      "type": "array",
      "description": "User-facing metric presets.",
      "items": { "$ref": "#/$defs/MetricProfile" }
    }
  },

  "$defs": {
    "MetricNode": {
      "$ref": "https://example.com/schema/metric-node.json"
    },

    "MetricPack": {
      "type": "object",
      "required": ["id", "name", "metricIds"],
      "properties": {
        "id": { "type": "string" },
        "name": { "type": "string" },
        "description": { "type": "string" },
        "metricIds": {
          "type": "array",
          "items": { "type": "string" }
        }
      }
    },

    "MetricDag": {
      "$ref": "https://example.com/schema/dag.json"
    },

    "MetricProfile": {
      "type": "object",
      "required": ["id", "name", "dagId"],
      "properties": {
        "id": { "type": "string" },
        "name": { "type": "string" },
        "description": { "type": "string" },
        "dagId": { "type": "string" },
        "configOverrides": {
          "type": "object",
          "additionalProperties": true
        }
      }
    }
  }
}
`
